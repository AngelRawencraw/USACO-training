#include <iostream>
#include <fstream>
#include <string>
using namespace std;
typedef long long int64;
int64 a[5000],b[5000];
void init(long l,long r)
{
                 long i,j,x,y;
                i=l;j=r;x=a[(i+j)/2];
                 do
                {
                                 while (a[i]<x) ++i;
                                 while (a[j]>x) --j;
                                 if (i<=j)
                                {
                                                y=a[i];a[i]=a[j];a[j]=y;
                                                y=b[i];b[i]=b[j];b[j]=y;
                                                i++;j--;
                                }
                }
                 while(i<=j);
                 if (i<r) init(i,r);
                 if (l<j) init(l,j);
}
int main()
{
                int64 mon=0,m,n;
                ofstream fout ( "milk.out",ios::out);
                ifstream fin ( "milk.in",ios::in);
                fin>>n>>m;
                 for(long i=1;i<=m;i++)
                                fin>>a[i]>>b[i];
                init(1,m); //排序
                 long j=0;
                 while (n>0)//贪心选择
                {
                                j++;
                                 if ((n>0)&&(n-b[j]>=0))
                                {
                                                n=n-b[j];
                                                mon=mon+a[j]*b[j];
                                }
                                 if ((n>0)&&(n-b[j]<0))
                                {
                                                mon=mon+a[j]*n;
                                                n=0;
                                }
                }
                fout<<mon<<endl;
    return 0;
}

[优秀程序]

#include <fstream>

#define MAXPRICE 1001

using namespace std;

 

int main() {

    ifstream fin ("milk.in");

    ofstream fout ("milk.out");

    unsigned int i, needed, price, paid, farmers, amount, milk[MAXPRICE][2];

    paid = 0;

    fin>>needed>>farmers;

    for(i = 0;i<farmers;i++){

        fin>>price>>amount;

        milk[price][0] += amount;  

    }

    for(i = 0; i<MAXPRICE && needed;i++){

        if(needed> = milk[i][0]) {

            needed -= milk[i][0];

            paid += milk[i][0] * i;

        } else if(milk[i][0]>0) {

            paid += i*needed;

            needed = 0;    

        }

    }

    fout << paid << endl;

    return 0;

}
