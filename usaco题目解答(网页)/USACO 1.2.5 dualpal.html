bool panduan(char *b,int le)
{
                 for(int i=0;i<(le+1)/2;i++)
                {
                                 if (b[i]!=b[le-i]) return false;
                }
                 return true ;
}

{
                 int num=0,s=0,ji=0;
                fin>>num>>s;

                                ++s;
                                 int time=0,jinzhi=10;
                                 while ((time<2)&&(jinzhi>=2))
                                {

void zhuanhuan(int a,int n)//任意数制转换函数
{
                 int m,r;
                m=r=-1;
                 while (a!=0)
                {
                                m=a%n;
                                r++;
                                 if (m<10) te[r]=m+48;
                                 else te[r]=m+55;
                                a=a/n;
                }
                le=r;
}
